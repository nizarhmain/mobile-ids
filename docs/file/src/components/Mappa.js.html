    src/components/Mappa.js

    import React, { Component } from 'react';
import { View, StyleSheet } from 'react-native';
import { MapView } from 'expo';
import Descriptor from './Descriptor';
import data from '../fakeData/markers.json';

export default class Mappa extends Component {
	//TODO hardcoded stuff here to clear up

	// lat and long passed by the locator component from upwards
	state = {
		mapRegion: { 
		latitude: this.props.location.coords.latitude,
		longitude: this.props.location.coords.longitude,
		latitudeDelta: 0.0922,
		longitudeDelta: 0.0421 
		},
			
		spotTitle: 'this is a generic spot'
		};

	componentWillMount() {
		this.setState({ markers: data });
	}


	handleMapRegionChange = mapRegion => {
		this.setState({ mapRegion });
	};

	fromMarkerToSpot(title, info, picture) {
		this.setState({ spotTitle: title });
		this.setState({ spotInfo: info });
		this.setState({ spotPicture: picture });
	}


	render() {
		return (
			<View style={styles.container}>
				<MapView
				style={{ alignSelf: 'stretch', height: 260 }}
				region={this.state.mapRegion}
				onRegionChange={this.handleMapRegionChange}
				showsPointsOfInterest={false}
				showsBuildings={false}
				showsIndoor={false}
				showsTraffic={false}
				toolbarEnabled={false}
				>
	
				{this.state.markers.map(marker => (
					<MapView.Marker 
					key={marker.title}
					coordinate={marker.coordinates}
					title={marker.title}
					image={marker.image}
					picture={marker.picture}
					pinColor={marker.pinColor}
					onPress={() => this.fromMarkerToSpot(marker.title, marker.info, marker.picture)}
					/>
				))}
					
				</MapView>

				<Descriptor title={this.state.spotTitle} info={this.state.spotInfo} picture={this.state.spotPicture} />
				
			</View>
		);
	}
}

const styles = StyleSheet.create({
	container: {
		flex: 1,
		alignItems: 'center',
		justifyContent: 'center'
	},
});
